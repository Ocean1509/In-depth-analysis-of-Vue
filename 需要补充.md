Reflect.ownKeys()


```
var obj = {
  a: 1,
  [Symbol('b')]: 1
}

console.log(Object.keys(obj)) // ['a']
console.log(Object.getOwnPropertyNames(obj)) // ['a']
console.log(Object.getOwnPropertySymbols(obj)) // [Symbol('b')]
console.log(Reflect.ownKeys(obj)) //  ["a", Symbol(b)]

```
